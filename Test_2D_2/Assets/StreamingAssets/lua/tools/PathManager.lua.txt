--[[
    封装了一些路径相关
--]]
local PathManager={}

function PathManager:Init()

    ---自动切换路径
    if IS_RELEASE_MODE then
        self.ResourcePath={
            --Bullet_1="XXXX.ab",
            --Character_1="XXXX.ab",
        }
    else
        self.ResourcePath={
            Bullet_1="Prefabs/Bullet",
            Character_1="Prefabs/Ruby",
        }
    end

end

---获取文件名
function PathManager:GetName(path)
    --匹配从 / 到 . 之间的字母、数字、下划线
    local reg = [[/([%w_]+)%.]]
    local name = string.match(path, reg)
    return name;
end

function PathManager:GetAssetName(go)

end

---返回值1: 去掉后缀名的字串
---返回值2: .后缀名
function PathManager:RemoveSuffix(path)
    local index = string.find(path, [[%.]])
    return string.sub(path, 1, index-1), string.sub(path, index)
end

PathManager:Init()

return PathManager
